        function formatDateJJMMAAAA(dateValue) {

            if (!dateValue) return '';

            if (typeof dateValue === 'object' && dateValue instanceof Date) {

                const jour = String(dateValue.getDate()).padStart(2, '0');

                const mois = String(dateValue.getMonth() + 1).padStart(2, '0');

                const annee = dateValue.getFullYear();

                return `${jour}/${mois}/${annee}`;

            }

            if (typeof dateValue === 'string' && dateValue.includes('Date(')) {

                const match = dateValue.match(/Date\((\d+),(\d+),(\d+)\)/);

                if (match) {

                    const annee = match[1];

                    const mois = String(parseInt(match[2], 10) + 1).padStart(2, '0');

                    const jour = String(match[3]).padStart(2, '0');

                    return `${jour}/${mois}/${annee}`;

                }

            }

            if (typeof dateValue === 'string' && dateValue.includes('-')) {

                const parts = dateValue.split('-');

                if (parts.length === 3) {

                    return `${parts[2]}/${parts[1]}/${parts[0]}`;

                }

            }

            return String(dateValue);

        }

        async function loadCours() {

            const content = document.getElementById('cours-content');

            content.innerHTML = '<div class="loading">Chargement des cours...</div>';

            try {

                const url = `https://docs.google.com/spreadsheets/d/${CONFIG.SHEET_COURS.ID}/gviz/tq?tqx=out:json&sheet=${CONFIG.SHEET_COURS.NAME}`;

                const response = await fetch(url);

                const text = await response.text();

                const jsonData = JSON.parse(text.substring(47, text.length - 2));

                processCoursData(jsonData);

            } catch (error) {

                content.innerHTML = `

                    <div class="error">

                        ❌ Erreur de chargement des cours<br>

                        <small>${error.message}</small>

                    </div>

                `;

            }

        }


        function processCoursData(data) {

            const rows = data.table.rows;

            const cols = data.table.cols;

            const headers = cols.map(col => col.label);

            

            const dataPoints = rows.map((row, index) => {

                const obj = { index };

                row.c.forEach((cell, i) => {

                    if (!cell) return;

                    const label = headers[i];

                    obj[label] = cell.v;

                    const norm = normalizeHeader(label);

                    if (norm && norm !== label) obj[norm] = cell.v;

                });

                return obj;

            }).filter(row => row[headers[0]] || row[normalizeHeader(headers[0])]);


            displayCours(dataPoints, headers);

        }

        function processPortfolioData(data) {

            const rows = data.table.rows;

            const cols = data.table.cols;

            const headers = cols.map(col => col.label);

            

            const portfolio = rows.map(row => {

                const obj = {};

                row.c.forEach((cell, i) => {

                    if (cell) obj[headers[i]] = cell.v;

                });

                return obj;

            }).filter(row => row[headers[0]]);


            displayPortfolio(portfolio, headers);

        }

        function createRepartitionChart(data, canvasId) {

            const actionsOnly = data.filter(row => {

                const ticker = row[Object.keys(row)[0]];

                return ticker && !ticker.toString().toUpperCase().includes('MOUCHARAKA') && !ticker.toString().toUpperCase().includes('OPCVM');

            });

            

            if (actionsOnly.length === 0) return;

            

            const labels = [];

            const values = [];

            const backgroundColors = [

                '#2E7D32',

                '#1565C0',

                '#C62828',

                '#F57C00',

                '#6A1B9A',

                '#00796B',

                '#E64A19',

                '#5D4037',

                '#455A64',

                '#AD1457'

            ];

            

            let totalValorisation = 0;

            

            actionsOnly.forEach(row => {

                const ticker = row[Object.keys(row)[0]];

                const nb = parseFloat(row['Nb'] || row['Quantité'] || row['Qte'] || 0);

                const cours = parseFloat(row['Cours.Act'] || row['Cours'] || row['Cours Actuel'] || 0);

                const valorisation = nb * cours;

                

                if (valorisation > 0) {

                    labels.push(ticker);

                    values.push(valorisation);

                    totalValorisation += valorisation;

                }

            });

            

            const percentages = values.map(v => ((v / totalValorisation) * 100).toFixed(1));

            

            const ctx = document.getElementById(canvasId);

            if (ctx) {

                new Chart(ctx, {

                    type: 'pie',

                    data: {

                        labels: labels.map((label, i) => `${label} (${percentages[i]}%)`),

                        datasets: [{

                            data: values,

                            backgroundColor: backgroundColors.slice(0, labels.length),

                            borderColor: '#fff',

                            borderWidth: 2,

                            hoverOffset: 10

                        }]

                    },

                    options: {

                        responsive: true,

                        maintainAspectRatio: false,

                        plugins: {

                            legend: {

                                position: 'right',

                                labels: {

                                    font: { size: 11 },

                                    padding: 15,

                                    generateLabels: function(chart) {

                                        const data = chart.data;

                                        return data.labels.map((label, i) => ({

                                            text: label,

                                            fillStyle: data.datasets[0].backgroundColor[i],

                                            hidden: false,

                                            index: i

                                        }));

                                    }

                                }

                            },

                            tooltip: {

                                callbacks: {

                                    label: function(context) {

                                        const value = context.parsed;

                                        const percentage = ((value / totalValorisation) * 100).toFixed(1);

                                        return `${context.label.split('(')[0]}: ${value.toLocaleString('fr-FR', {minimumFractionDigits: 2})} MAD (${percentage}%)`;

                                    }

                                }

                            }

                        }

                    }

                });

            }
